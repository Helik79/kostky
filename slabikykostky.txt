    <script>
        // Definice písmen pro rotaci
        const consonants = ['M', 'S', 'L', 'P', 'M', 'S']; // 6 stěn
        const vowels = ['A', 'Á', 'E', 'É', 'I', 'Í', 'O', 'Ó', 'U', 'Ů'];

        // Definice 3D rotací pro jednotlivé stěny (indexy 0 až 5)
        const rotations = [
            { x: 0, y: 0 },       // přední
            { x: 0, y: 180 },     // zadní
            { x: -90, y: 0 },     // horní
            { x: 90, y: 0 },      // dolní
            { x: 0, y: 90 },      // levá
            { x: 0, y: -90 }      // pravá
        ];

        function rollDice() {
            const cube1 = document.getElementById('cube1');
            const cube2 = document.getElementById('cube2');

            // Vybereme náhodnou stěnu (0-5)
            const side1 = Math.floor(Math.random() * 6);
            const side2 = Math.floor(Math.random() * 6);

            // Dynamicky změníme písmena na pravé kostce, aby se vystřídalo všech 10 samohlásek
            const cube2Faces = cube2.getElementsByClassName('face');
            // Zamícháme samohlásky a přiřadíme je stěnám před otočením
            const shuffledVowels = [...vowels].sort(() => 0.5 - Math.random());
            for(let i = 0; i < 6; i++) {
                cube2Faces[i].innerText = shuffledVowels[i];
            }

            // Přidáme extra otočky (360° * 2), aby animace vypadala jako opravdové točení
            const extraTurnsX1 = (Math.floor(Math.random() * 3) + 2) * 360;
            const extraTurnsY1 = (Math.floor(Math.random() * 3) + 2) * 360;
            const extraTurnsX2 = (Math.floor(Math.random() * 3) + 2) * 360;
            const extraTurnsY2 = (Math.floor(Math.random() * 3) + 2) * 360;

            // Aplikujeme rotaci
            cube1.style.transform = `rotateX(${rotations[side1].x + extraTurnsX1}deg) rotateY(${rotations[side1].y + extraTurnsY1}deg)`;
            cube2.style.transform = `rotateX(${rotations[side2].x + extraTurnsX2}deg) rotateY(${rotations[side2].y + extraTurnsY2}deg)`;
        }

        // První automatické hození po načtení stránky
        window.onload = rollDice;
    </script>
